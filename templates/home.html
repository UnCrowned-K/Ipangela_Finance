from pulp import *
# Define optimization problem
prob = LpProblem("Maximize_Profit", LpMaximize)
# Decision variables
x = LpVariable("x", lowBound=0, cat='Integer')
y = LpVariable("y", lowBound=0, cat='Integer')
# Objective function
prob += 45*x + 80*y
# Constraints
prob += x + y <= 150
